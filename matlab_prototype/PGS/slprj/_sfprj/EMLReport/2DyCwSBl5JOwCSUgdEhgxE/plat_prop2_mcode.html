1function [anga,angb,a,b,lmin] = plat_prop()
 2rb = .5;                        % Radio de la junta en la base
 3ra = .3;                        % Radio de la junta en la plataforma
 4lmin = 2;                       % Extensión minima del piston/actuador
 5hb = .1;                        % Separación lineal entre juntas en la base
 6ha = .1;                        % Separacion lineal entre juntas en la plataforma
 7% Iniciar los valores de la plataforma
 8
 9th_b = asin(hb/(2*rb));
10th_a = asin(ha/(2*ra));
11
12angb = zeros(6,1);
13anga = zeros(6,1);
14for i = 1:6
15  anga(i,1) = ((2*pi)/3)*floor((i+1)/2) - (pi/3) + (-1)^i*th_a;
16  angb(i,1) = ((2*pi)/3)*floor(i/2) - (-1)^i*th_b;
17end
18
19a = zeros(3,6);
20b = zeros(3,6);
21for i = 1:6
22  a(1,i) = ra*cos(anga(i));
23  a(2,i) = ra*sin(anga(i));
24  a(3,i) = 0;
25  b(1,i) = rb*cos(angb(i));
26  b(2,i) = rb*sin(angb(i));
27  b(3,i) = 0;
28end
29end
