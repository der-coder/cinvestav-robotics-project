1function R = M_Rot(ang,eje,tipo)
 2%Se obtiene la matriz de rotacion en el eje x, y o z según se especifique
 3%tanto en grados como en radianes.
 4%
 5%   R = M_Rot(ang,ejer,tipo)
 6% 
 7%   ang  - ángulo de rotación
 8%   eje  - eje de la rotación 'x','y' o 'z'
 9%   tipo - tipo de dato del ángulo 'g' -> grados 'r' -> radianes
10
11switch nargin
12  case 1
13    disp("Faltan datos")
14  case 2
15    if strcmp(eje,'x')
16      R = rotx(ang);
17    end
18    if strcmp(eje,'y')
19      R = roty(ang);
20    end
21    if strcmp(eje,'z')
22      R = rotz(ang);
23    end
24  case 3
25    if strcmp(tipo,'g')
26      ang = deg2rad(ang);
27    end
28    if strcmp(eje,'x')
29      R = rotx(ang);
30    end
31    if strcmp(eje,'y')
32      R = roty(ang);
33    end
34    if strcmp(eje,'z')
35      R = rotz(ang);
36    end
37end
38
39  function Rx = rotx(theta)
40    Rx = [1,0,0;
41      0,cos(theta),-sin(theta);
42      0,sin(theta),cos(theta)];
43  end
44  function Ry = roty(theta)
45    Ry = [cos(theta),0,sin(theta);
46      0,1,0;
47      -sin(theta),0,cos(theta)];
48  end
49  function Rz = rotz(theta)
50    Rz = [cos(theta),-sin(theta),0;
51      sin(theta),cos(theta),0;
52      0,0,1];
53  end
54
55end
